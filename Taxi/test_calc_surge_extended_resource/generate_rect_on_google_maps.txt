"""
Code helps to visualize 4 splitted request on Google Maps

For debug and future tests

Links to draw rectangles:
https://developers.google.com/maps/documentation/javascript/examples/rectangle-simple
https://jsfiddle.net/api/post/library/pure/
"""


rects = [
    [(9.990879, 10.009041), (10.009121, 9.990959)],
    [(9.990879, 10.009041), (10.000912, 9.999102)],
    [(9.999088, 10.009041), (10.009121, 9.999102)],
    [(9.990879, 10.000898), (10.000912, 9.990959)],
    [(9.999088, 10.000898), (10.009121, 9.990959)],
]

total = """

// This example adds a red rectangle to a map.
function initMap() {
  const map = new google.maps.Map(document.getElementById("map"), {
    zoom: 13,
    center: { lat: 10.00, lng: 10.00 },
    mapTypeId: "terrain",
  });
  
  new google.maps.Marker({
    position: { lat: 10.0, lng: 10.0 },
    map,
    title: "query!",
  });
    
"""

for i in range(len(rects)):
    rect = rects[i]
    clr = '#FF0000'
    if i == 0:
        clr = '#FFFFFF'

    code = (
        """ const rectangle"""
        + str(i)
        + """  = new google.maps.Rectangle({
        strokeColor: "#FFFFFF",
        strokeOpacity: 0.8,
        strokeWeight: 2,
        fillColor: \""""
        + clr
        + """\",
        fillOpacity: 0.35,
        map,
        bounds: { 
        """
    )

    code += ' north: ' + str(rect[0][1]) + ',\n'
    code += ' south: ' + str(rect[1][1]) + ',\n'
    code += ' east: ' + str(rect[1][0]) + ',\n'
    code += ' west: ' + str(rect[0][0]) + ',\n'

    code += """
    },
    });
    """

    total += code + '\n'

total += '\n\n\n}'

print(total)
